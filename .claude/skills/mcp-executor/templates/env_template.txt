# ============================================================================

# Copy this file to .env and edit with your values

# ============================================================================
# EMAIL MCP SERVER CONFIGURATION
# ============================================================================

# Choose email backend: 'gmail' (default, OAuth) or 'smtp' (universal)
# Default: gmail
EMAIL_BACKEND=gmail

# ============================================================================
# OPTION 1: GMAIL API BACKEND (OAuth 2.0 - Secure)
# ============================================================================
# Use this if you want secure Gmail API integration
# Requires OAuth 2.0 credentials from Google Cloud Console

# Path to credentials.json downloaded from Google Cloud Console
# Get it from: https://console.cloud.google.com/credentials
# REQUIRED for EMAIL_BACKEND=gmail
GMAIL_CREDENTIALS_FILE=credentials.json

# Path to token.json (auto-created on first OAuth flow)
# This file stores the OAuth token and is refreshed automatically
GMAIL_TOKEN_FILE=token.json

# ============================================================================
# OPTION 2: SMTP BACKEND (TLS - Works with Any Email Provider)
# ============================================================================
# Use this if you want SMTP integration with any email provider

# SMTP server hostname
# Examples:
#   Gmail: smtp.gmail.com
#   Outlook: smtp.office365.com
#   Yahoo: smtp.mail.yahoo.com
#   Custom: mail.example.com
# REQUIRED for EMAIL_BACKEND=smtp
SMTP_HOST=smtp.gmail.com

# SMTP server port
# Common: 587 (TLS encryption), 465 (SSL encryption)
# Default: 587
SMTP_PORT=587

# SMTP authentication username (email address)
# Example: user@gmail.com
# REQUIRED for EMAIL_BACKEND=smtp
SMTP_USERNAME=your_email@gmail.com

# SMTP authentication password
# For Gmail: Use app-specific password (not your regular password)
#   Get it from: https://myaccount.google.com/apppasswords
# For Outlook: Use your password or app password
# For Yahoo: Use app password from https://login.yahoo.com/account/security
# REQUIRED for EMAIL_BACKEND=smtp
SMTP_PASSWORD=your_app_specific_password_here

# SMTP from address (optional, defaults to SMTP_USERNAME)
# Use this if you want emails to come from a different address
# Example: notifications@company.com
SMTP_FROM_ADDRESS=your_email@gmail.com

EXECUTOR_MAX_RETRIES=3

#   1. Set EMAIL_BACKEND=smtp

#   4. Set SMTP_USERNAME=your@outlook.com
#   5. Set SMTP_PASSWORD=your_password

# SMTP (Custom):

#   3. Set SMTP_PORT=587 (or 465 for SSL)

#   5. Set SMTP_PASSWORD=company_password
